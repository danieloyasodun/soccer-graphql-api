Basic Queries:

Get all available competitions and seasons
query {
  competitions
  seasons
}

Search for a specific player
query {
  searchPlayers(query: "Messi") {
    id
    name
    nation
    position
    birthYear
  }
}

Search for a team
query {
  searchTeams(query: "Barcelona") {
    id
    name
    fbrefUrl
  }
}

Complex Nested Query

query CompleteAnalysis {
  # Get top team
  topTeam: teamLeaderboard(
    seasonEndYear: 2024
    competition: "Premier League"
    sortBy: GOALS
    limit: 1
  ) {
    name
    seasonStats(seasonEndYear: 2024, competition: "Premier League") {
      goals
      xG
      possessionPct
    }
    
    # Get their top scorer
    players(seasonEndYear: 2024) {
      name
      seasonStats(seasonEndYear: 2024, competition: "Premier League") {
        goals
        assists
        xG
      }
    }
  }
  
  # Get top scorer in the league
  topScorer: topScorers(
    seasonEndYear: 2024
    competition: "Premier League"
    limit: 1
  ) {
    name
    currentTeam {
      name
    }
    careerStats {
      totalGoals
      seasonsPlayed
    }
  }
  
  # Get all competitions
  competitions
}

  "limit": 10
